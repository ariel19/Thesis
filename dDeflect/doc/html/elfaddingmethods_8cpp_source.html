70  const QList<Elf64_Addr> &except_list) {

72  if (data_size % addr_size)
73  return false;
74 
75  quint64 addr;
76 
77  for (int i = 0; i < data_size; i += addr_size) {
78  addr = 0;
79  for (int j = 0; j < addr_size; ++j)
80  addr |= (static_cast<const char>((addr_data.data())[i + j]) & 0xff) << (j * 8);
81  if (!except_list.contains(addr))
82  addr_list.push_back(addr);

84
