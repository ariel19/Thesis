119 bool ELFAddingMethods::wrapper_gen_code(Wrapper<RegistersType> *wrap, QString &code) {
120  if (!wrap)
121  return false;
122 
123  // check if ret is in used registers (if it is remove)
124  if (wrap->used_regs.indexOf(wrap->ret) != -1)
125  wrap->used_regs.removeAll(wrap->ret);
126 
127  // generate push registers
128  code.append(AsmCodeGenerator::push_regs<RegistersType>(wrap->used_regs));
129 
130  code.append(wrap->code);
131  // fill params
132  uint64_t filled_params = fill_params(code, wrap->static_params);
133 
134  // generate pop registers
135  QList<RegistersType> rused_args;
136  rused_args.reserve(wrap->used_regs.size());
137  std::reverse_copy(wrap->used_regs.begin(), wrap->used_regs.end(), std::back_inserter(rused_args));
138 
139  code.append(AsmCodeGenerator::pop_regs<RegistersType>(rused_args));
140 
141  return true;
142 }
