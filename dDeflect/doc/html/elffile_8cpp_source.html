45  // pad data
46  for (i = 0; i < dsize; ++i)
47  data.append('\x00');

587 template <typename ElfProgramHeaderType, typename ElfOffsetType>
588 bool ELF::__find_pre_pad(const ElfProgramHeaderType *ph, const ElfProgramHeaderType *phn,
589  const int dsize, uint32_t *pre_pad) {
590 
591  uint8_t align = sizeof(ElfOffsetType);
592 
593  if (!ph)
594  return false;
595 
596  try {
597  *pre_pad = align - ((ph->p_vaddr + ph->p_memsz) % align ? : align);
598 
599  // 1. if there is no next LOAD segment after current
600  // 2. enough space between end of current segment in memory and next segment in memory
601  if (!(!phn || __round_address_down(ph->p_vaddr + ph->p_memsz + (*pre_pad) + dsize,
602  ph->p_align) < __round_address_down(phn->p_vaddr, phn->p_align)))
603  return false;
604  // pad if size of file and memory images are different
605  *pre_pad += (ph->p_memsz - ph->p_filesz);
606  }
607  catch(const std::exception &) {
608  return false;
609  }
610  return true;
611 
612  return true;
613 }

615 template <typename ElfProgramHeaderType, typename ElfOffsetType>
616 bool ELF::__find_post_pad(const ElfProgramHeaderType *ph, const ElfProgramHeaderType *phn,
617  const int dsize, const uint32_t pre_pad, uint32_t *post_pad,
618  bool *change_vma) {
619 
620  uint8_t align = sizeof(ElfOffsetType);
621  if (!ph)
622  return false;
623 
624  try {
625  // pad next segment in file if exists to vaddr % align == offset % align
626  if (phn) {
627  *post_pad = phn->p_align + (phn->p_offset % phn->p_align) -
628  ((phn->p_offset + dsize + pre_pad) % phn->p_align);
629  }
630  // if next segment is absebt we need to pad size of new data to the size of pointer
631  // need to increase a highest vaddr in file, because there is no next segment in file
632  else {
633  *post_pad = align - ((pre_pad + dsize + *post_pad) % align ? : align);
634  *change_vma = true;
635  }
636  }
637  catch(const std::exception &) {
638  return false;
639  }
640  return true;

652 std::pair<QByteArray, Elf64_Addr> ELF::__construct_data(const QByteArray &data, ELF::best_segment &bs) {
653  static std::pair<QByteArray, Elf64_Addr> failed(QByteArray(), 0);
654  if (!parsed)
655  return failed;
656 
657  uint32_t total_space = data.size() + bs.post_pad + bs.pre_pad;
658  std::pair<ex_offset_t, ex_offset_t> fo_va;
659  ex_offset_t file_off = 0, va = 0;
660 
661  switch(cls) {
662  case classes::ELF32:
663  fo_va = __get_new_data_va_fo<Elf32_Phdr>(bs);
664  break;
665  case classes::ELF64:
666  fo_va = __get_new_data_va_fo<Elf64_Phdr>(bs);
667  break;
668  default:
669  return std::make_pair(QByteArray(), 0);
670  }
671 
672  file_off = fo_va.first;
673  va = fo_va.second;
674 
675  // 1. copy data from part of file, till new offset part
676  // 2. copy pre_pad size, data, post_pad size
677  // 3. copy rest of data from file
678  QByteArray new_b_data(b_data.data(), file_off);
679  for (uint32_t i = 0; i < bs.pre_pad; ++i)
680  new_b_data.append('\0');
681  new_b_data.append(data);
682  for (uint32_t i = 0; i < bs.post_pad; ++i)
683  new_b_data.append('\0');
684  new_b_data.append(b_data.data() + file_off, b_data.size() - file_off);
685 
686  // fixing...
687  Elf64_Addr vaddr;
688  switch(cls) {
689  case classes::ELF32:
690  __fix_elf_header<Elf32_Ehdr>(new_b_data, file_off, total_space);
691  __fix_section_table<Elf32_Ehdr, Elf32_Shdr>(new_b_data, file_off, total_space);
692  vaddr = __fix_segment_table<Elf32_Phdr>(new_b_data, file_off, total_space, bs.pre_pad + data.size());
693  __fix_vma<Elf32_Dyn, Elf32_Sym, Elf32_Word>(new_b_data, bs, file_off, vaddr);
694  break;
695  case classes::ELF64:
696  __fix_elf_header<Elf64_Ehdr>(new_b_data, file_off, total_space);
697  __fix_section_table<Elf64_Ehdr, Elf64_Shdr>(new_b_data, file_off, total_space);
698  vaddr = __fix_segment_table<Elf64_Phdr>(new_b_data, file_off, total_space, bs.pre_pad + data.size());
699  __fix_vma<Elf64_Dyn, Elf64_Sym, Elf64_Word>(new_b_data, bs, file_off, vaddr);
700  break;
701  default:
702  return failed;
703  }
704 
705  return std::make_pair(new_b_data, va);
706 }

739  const uint32_t insert_space, const uint32_t payload_size) {
740  Elf64_Addr va = 0;
741 
742  ElfProgramHeaderType *ph = nullptr;
743  foreach (ex_offset_t fo, ph_idx) {
744  ph = reinterpret_cast<ElfProgramHeaderType*>(data.data() + fo);
745  if (!ph)
746  return 0;
747 
748  if (ph->p_offset >= file_off)
749  ph->p_offset += insert_space;
750 
751  // check if current one is extended segment
752  if (ph->p_type == PT_LOAD && ph->p_offset + ph->p_filesz == file_off) {
753  ph->p_filesz += payload_size;
754  ph->p_memsz = ph->p_filesz;
755  // set executable flag on segment (may provide to vulnerabilities)
756  // TODO: fuck of W
757  ph->p_flags |= PF_X;
758 
759  va = ph->p_vaddr + ph->p_memsz;
760  }
761  }
762 
763  return va;
