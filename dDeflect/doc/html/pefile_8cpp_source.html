281 uint64_t PEFile::injectUniqueData(QByteArray data, QMap<QByteArray, uint64_t> &ptrs, bool *inserted)
282 {
283  QByteArray hash = QCryptographicHash::hash(data, QCryptographicHash::Sha3_512);
284 
285  if(inserted)
286  *inserted = false;
287 
288  if(ptrs.contains(hash))
289  return ptrs[hash];
290 
291  unsigned int fileOffset = 0;
292  unsigned int memOffset = 0;
293  bool is_added = false;
294 
295  // Próba dodania kodu do każdej z sekcji
296  for(unsigned int i = 0; !is_added && i < numberOfSections; ++i)
297  is_added = addDataToSection(i, data, fileOffset, memOffset);
298 
299  // Próba dodania kodu do każdej z sekcji z rozszerzeniem
300  for(unsigned int i = 0; !is_added && i < numberOfSections - 1; ++i)
301  is_added = addDataToSection(i, data, fileOffset, memOffset);
302 
303  // Tworzenie nowej sekcji, lub rozszerzanie ostatniej
304  std::uniform_int_distribution<int> prob(0, 9);
305  if(prob(gen) == 0)
306  {
307  if(!is_added)
308  is_added = addNewSection(getRandomSectionName(), data, fileOffset, memOffset);
309 
310  if(!is_added)
311  is_added = resizeLastSection(data, fileOffset, memOffset);
312  }
313  else
314  {
315  if(!is_added)
316  is_added = resizeLastSection(data, fileOffset, memOffset);
317 
318  if(!is_added)
319  is_added = addNewSection(getRandomSectionName(), data, fileOffset, memOffset);
320  }
321 
322  if(!is_added)
323  return 0;
324 
325  uint64_t offset = memOffset + getImageBase();
326  ptrs.insert(hash, offset);
327  if(inserted)
328  *inserted = true;
329 
330  return offset;
331 }

361 bool PEFile::addRelocations(QList<uint64_t> relocations)
362 {
363  if(!parsed)
364  return false;
365 
366  if(getRelocationsSize() == 0)
367  return true;
368 
369  QList<RelocationTable> reloc_table;
370  if(!getRelocations(reloc_table))
371  return false;
372 
373  // Dodawanie do tablicy
374  foreach(uint64_t abs_addr, relocations)
375  {
376  uint32_t rel_addr = abs_addr - getImageBase();
377  uint32_t va = rel_addr & 0xFFFFF000;
378  uint32_t offset = rel_addr & 0x00000FFF;
379 
380  QList<RelocationTable> new_reloc_table;
381  bool added = false;
382 
383  for(QList<RelocationTable>::iterator it = reloc_table.begin(); it != reloc_table.end(); ++it)
384  {
385  if(it->VirtualAddress < va || (added && it->VirtualAddress > va))
386  {
387  new_reloc_table.append(*it);
388  continue;
389  }
390 
391  if(it->VirtualAddress == va)
392  {
393  RelocationTable r = *it;
394  r.addOffset(offset, getRelocationType());
395  new_reloc_table.append(r);
396 
397  added = true;
398  }
399 
400  if(!added && it->VirtualAddress > va)
401  {
402  RelocationTable r;
403  r.VirtualAddress = va;
404  r.SizeOfBlock = IMAGE_SIZEOF_BASE_RELOCATION;
405  r.addOffset(offset, getRelocationType());
406  new_reloc_table.append(r);
407 
408  added = true;
409  new_reloc_table.append(*it);
410  }
411  }
412 
413  if(!added)
414  {
415  RelocationTable r;
416  r.VirtualAddress = va;
417  r.SizeOfBlock = IMAGE_SIZEOF_BASE_RELOCATION;
418  r.addOffset(offset, getRelocationType());
419  new_reloc_table.append(r);
420  }
421 
422  reloc_table = new_reloc_table;
423  }
424 
425  QByteArray raw_table;
426  foreach(RelocationTable rt, reloc_table)
427  raw_table.append(rt.toBytes());
428 
429  unsigned int file_offset, mem_offset;
430  if(!addNewSection(getRandomSectionName(), raw_table, file_offset, mem_offset))
431  return false;
432 
433  getDataDirectory(IMAGE_DIRECTORY_ENTRY_BASERELOC)->Size = raw_table.length();
434  getDataDirectory(IMAGE_DIRECTORY_ENTRY_BASERELOC)->VirtualAddress = mem_offset;
435 
436  return true;
437 }

522 {

535  while(i < getRelocationsSize())

538

541  t.VirtualAddress = reloc_tab.VirtualAddress;
542 
543  printf("%x: %d\n", t.VirtualAddress, t.SizeOfBlock);
544 
545  uint32_t j = IMAGE_SIZEOF_BASE_RELOCATION;
546 
547  while(j < t.SizeOfBlock)
548  {
549  RelocationTable::TypeOffset typeOffset;

618 bool PEFile::parse()
619 {
620  char *data = b_data.data();
621  size_t length = b_data.length();
622 
623  PIMAGE_DOS_HEADER dosHeader = reinterpret_cast<PIMAGE_DOS_HEADER>(data);
624  dosHeaderIdx = 0;
625 
626  if(length < sizeof(IMAGE_DOS_HEADER) || dosHeader->e_magic != IMAGE_DOS_SIGNATURE)
627  return false;
628 
629  _is_x64 = isPE_64(dosHeader->e_lfanew);
630 
631  PIMAGE_NT_HEADERS32 ntHeaders32 = NULL;
632  PIMAGE_NT_HEADERS64 ntHeaders64 = NULL;
633 
634  if(_is_x64)
635  {
636  ntHeaders64 = reinterpret_cast<PIMAGE_NT_HEADERS64>(reinterpret_cast<char*>(dosHeader) +
637  dosHeader->e_lfanew);
638  ntHeadersIdx = reinterpret_cast<char*>(ntHeaders64) - reinterpret_cast<char*>(dosHeader);
639 
640  if(length < dosHeader->e_lfanew + sizeof(IMAGE_NT_HEADERS64::Signature)
641  || getNtHdrSignature() != IMAGE_NT_SIGNATURE)
642  return false;
643  }
644  else
645  {
646  ntHeaders32 = reinterpret_cast<PIMAGE_NT_HEADERS32>(reinterpret_cast<char*>(dosHeader) +
647  dosHeader->e_lfanew);
648  ntHeadersIdx = reinterpret_cast<char*>(ntHeaders32) - reinterpret_cast<char*>(dosHeader);
649 
650  if(length < dosHeader->e_lfanew + sizeof(IMAGE_NT_HEADERS32::Signature)
651  || getNtHdrSignature() != IMAGE_NT_SIGNATURE)
652  return false;
653  }
654 
655  PIMAGE_FILE_HEADER fileHeader = _is_x64 ?
656  reinterpret_cast<PIMAGE_FILE_HEADER>(&ntHeaders64->FileHeader) :
657  reinterpret_cast<PIMAGE_FILE_HEADER>(&ntHeaders32->FileHeader);
658 
659  fileHeaderIdx = reinterpret_cast<char*>(fileHeader) - reinterpret_cast<char*>(dosHeader);
660 
661  if(length < (fileHeaderIdx + sizeof(IMAGE_FILE_HEADER)))
662  return false;
663 
664  PIMAGE_OPTIONAL_HEADER32 optionalHeader32 = NULL;
665  PIMAGE_OPTIONAL_HEADER64 optionalHeader64 = NULL;
666 
667  if(_is_x64)
668  optionalHeader64 = reinterpret_cast<PIMAGE_OPTIONAL_HEADER64>(&ntHeaders64->OptionalHeader);
669  else
670  optionalHeader32 = reinterpret_cast<PIMAGE_OPTIONAL_HEADER32>(&ntHeaders32->OptionalHeader);
671 
672  optionalHeaderSize = fileHeader->SizeOfOptionalHeader;
673  optionalHeaderIdx = _is_x64 ?
674  (reinterpret_cast<char*>(optionalHeader64) - reinterpret_cast<char*>(dosHeader)) :
675  (reinterpret_cast<char*>(optionalHeader32) - reinterpret_cast<char*>(dosHeader));
676 
677  if(length < optionalHeaderIdx + optionalHeaderSize)
678  return false;
679 
680  if(_is_x64 && optionalHeader64->Magic != IMAGE_NT_OPTIONAL_HDR64_MAGIC)
681  return false;
682 
683  if(!_is_x64 && optionalHeader32->Magic != IMAGE_NT_OPTIONAL_HDR32_MAGIC)
684  return false;
685 
686  // Data Directories
687  numberOfDataDirectories = getOptHdrNumberOfRvaAndSizes();
688 
689  if(dataDirectoriesIdx)
690  delete [] dataDirectoriesIdx;
691 
692  try
693  {
694  dataDirectoriesIdx = new unsigned int[numberOfDataDirectories];
695  }
696  catch(std::bad_alloc &)
697  {
698  dataDirectoriesIdx = NULL;
699  return false;
700  }
701

703  reinterpret_cast<char*>(&(optionalHeader64->DataDirectory[0])) - reinterpret_cast<char*>(dosHeader) :
